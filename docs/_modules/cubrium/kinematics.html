[docs]def gridvecns(F):
    "Normal and shear components of gridvectors (edges of cube)."

    Fn = np.zeros(3)
    Fs = np.zeros((3, 3))
    for i in range(3):
        Fn[i] = np.linalg.norm(F[:, i])

    Fs = F.copy()
    # np.fill_diagonal(Fs,np.nan)

    return Fn, Fs
